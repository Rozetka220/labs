#include <iostream>
#include <cstdlib>
#include <ctime>
#include <iomanip>
#include <clocale>

using namespace std;

int main()
{
	srand(time(0));
	setlocale(LC_ALL, "Rus");
	 int a, b, buff, max;
	 cout << "Введите размерность массива : " << endl;
	 cin >> a
		 >> b;
	int **mas = new int*[a];
	for (int i = 0; i < a; i++)
	{
		mas[i] = new int[b];
	}

	for (int i = 0; i < a; i++)
	{
		for (int j = 0; j < b; j++)
		{
			mas[i][j] =10 + rand()%(99 - 10 +1);
		}
	}
	cout << "Первичная матрица" << endl;
	for (int i = 0; i < a; i++)
	{
		for (int j = 0; j < b; j++)
		{
			cout << mas[i][j] << setw(4);
		}
		cout << endl;
	}

	for (int i = 0; i < a; i++)
	{
		max = mas[i][0];
		buff = 0;
		for (int j = 0; j < b; j++)
		{	
			if (mas[i][j] > max)
			{
				max = mas[i][j];
				buff = j;
			}
			if (j == b-1)
			{		
				mas[i][buff] = mas[i][0];
				mas[i][0] = max;
			}			
		}		
	}
	cout << "Измененная  матрица" << endl;
	for (int i = 0; i < a; i++)
	{
		for (int j = 0; j < b; j++)
		{
			cout << mas[i][j] << setw(4);
		}
		cout << endl;
	}
	for (int i = 0; i < a; i++)
	{
		delete[]mas[i];
	}
	delete[]mas;
	system("pause");	
}
