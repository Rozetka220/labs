    unsigned char dice[9][9] = { "########",//0
								 "#      #",//1
								 "#      #",//2

								 "#      #",//4
								 "#      #",//5
								 "#      #",//6

int b,p;

void Screen()
{
	for (int i = 0; i < 9; i++)
	{
		for (int j = 0; j < 9; j++)
		{
			cout << dice[i][j];
		}
		cout << endl;
	}
	cout << "\n \n" << endl;
}

void Cube()
{
	int a, i = 1;
	while (i <= 2)
	{
		a =1 + rand() % 6;;
		switch (a)
		{
		case 1:
			dice[3][4] = '*';
			break;
		case 2:
			dice[2][5] = '*';
			dice[4][3] = '*';
			break;
		case 3:
			dice[2][5] = '*';
			dice[4][3] = '*';
			dice[3][4] = '*';
			break;
		case 4:
			dice[2][5] = '*';
			dice[4][3] = '*';
			dice[2][3] = '*';
			dice[4][5] = '*';
			break;
		case 5:
			dice[2][5] = '*';
			dice[2][3] = '*';
			dice[4][3] = '*';
			dice[4][5] = '*';
			dice[3][4] = '*';
			break;
		case 6:
			dice[2][5] = '*';
			dice[4][3] = '*';
			dice[2][3] = '*';
			dice[4][5] = '*';
			dice[2][4] = '*';
			dice[4][4] = '*';
			break;
		default:
		{
			cout << "Что то не так" << endl;
		}
		}
		Screen();
		b += a;
		p += a;
		
		for (int n=0; n < 9; n++)
		{
			for (int m=0; m < 9; m++)
			{
				dice[n][m] = default_dice[n][m];
			}
		}
		i++;
	}
}



int main()
{

	FirstPlayer = rand() % 2;

		if (FirstPlayer)

			cout << "Выпала решка \n";
		
		}
